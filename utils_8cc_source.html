151 TEST_CASE("utilities [fit_fwhm]", "[fit_fwhm]") {
152  // testing that the gaussian fitting works.
153  t_int imsizex = 512;
154  t_int imsizey = 512;
155  Image<t_real> psf = Image<t_real>::Zero(imsizey, imsizex);
156  // choice of parameters
157  t_real const fwhm_x = 3;
158  t_real const fwhm_y = 6;
159  t_real const theta = 0;
160  // setting up Gaussian calculation
161  t_real const sigma_x = fwhm_x / (2 * std::sqrt(2 * std::log(2)));
162  t_real const sigma_y = fwhm_y / (2 * std::sqrt(2 * std::log(2)));
163  // calculating Gaussian
164  t_real const a = std::pow(std::cos(theta), 2) / (2 * sigma_x * sigma_x) +
165  std::pow(std::sin(theta), 2) / (2 * sigma_y * sigma_y);
166  t_real const b = -std::sin(2 * theta) / (4 * sigma_x * sigma_x) +
167  std::sin(2 * theta) / (4 * sigma_y * sigma_y);
168  t_real const c = std::pow(std::sin(theta), 2) / (2 * sigma_x * sigma_x) +
169  std::pow(std::cos(theta), 2) / (2 * sigma_y * sigma_y);
170  auto x0 = imsizex * 0.5;
171  auto y0 = imsizey * 0.5;
172  for (t_int i = 0; i < imsizex; ++i) {
173  for (t_int j = 0; j < imsizey; ++j) {
174  t_real x = i - x0;
175  t_real y = j - y0;
176  psf(j, i) = std::exp(-a * x * x + 2 * b * x * y - c * y * y);
177  }
178  }
179  auto const fit = utilities::fit_fwhm(psf, 3);
180  auto new_fwhm_x = std::get<0>(fit) * 2 * std::sqrt(2 * std::log(2));
181  auto new_fwhm_y = std::get<1>(fit) * 2 * std::sqrt(2 * std::log(2));
182  auto new_theta = std::get<2>(fit);
183  CHECK(std::abs(new_fwhm_x - fwhm_x) < 1e-13);
184  CHECK(std::abs(new_fwhm_y - fwhm_y) < 1e-13);
185  CHECK(std::abs(new_theta - theta) < 1e-13);
186 }
