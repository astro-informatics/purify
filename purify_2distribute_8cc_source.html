77 Vector<t_int> equal_distribution(Vector<t_real> const &u, Vector<t_real> const &v,
78  t_int const &grid_size) {
79  // distribute visibilities by density calculated from histogram
80  t_real const max_u = u.array().maxCoeff();
81  t_real const max_v = v.array().maxCoeff();
82  t_real const min_u = u.array().minCoeff();
83  t_real const min_v = v.array().minCoeff();
84  // scaling for histogram grid
85  Vector<t_real> const scaled_u = (u.array() - min_u) * (grid_size - 1) / (max_u - min_u);
86  Vector<t_real> const scaled_v = (v.array() - min_v) * (grid_size - 1) / (max_v - min_v);
87  Image<t_int> histogram = Image<t_int>::Zero(grid_size, grid_size);
88  Vector<t_int> index = Vector<t_int>::LinSpaced(u.size(), 0, u.size());
89  // creating histogram grid
90  for (t_int i = 0; i < index.size(); i++) {
91  histogram((t_int)std::floor(scaled_u(i)), (t_int)std::floor(scaled_v(i))) += 1;
92  }
93  // sorting indicies
94  std::sort(index.data(), index.data() + index.size(),
95  [&histogram, &scaled_u, &scaled_v](int a, int b) {
96  return (histogram((t_int)std::floor(scaled_u(a)), (t_int)std::floor(scaled_v(a))) <
97  histogram((t_int)std::floor(scaled_u(b)), (t_int)std::floor(scaled_v(b))));
98  });
99  return index;

117  PURIFY_DEBUG("clustering iteration {}", n);

121  const t_real cost_val = cost(w(i) - w_centre.at(node));
122  if (cost_val < min) {
123  min = cost_val;
124  w_node[i] = node;
125  }
126  }
127  w_sum[w_node.at(i)] += w(i);
128  w_count[w_node.at(i)]++;

138  }
