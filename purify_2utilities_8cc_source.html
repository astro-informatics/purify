370 t_real calculate_l2_radius(const t_uint y_size, const t_real &sigma, const t_real &n_sigma,
371  const std::string distirbution) {
372  /*
373  Calculates the epsilon, the radius of the l2_ball in sopt
374  y:: vector for the l2 ball
375  */
376  if (distirbution == "chi^2") {
377  if (sigma == 0) {
378  return std::sqrt(2 * y_size + n_sigma * std::sqrt(4 * y_size));
379  }
380  return std::sqrt(2 * y_size + n_sigma * std::sqrt(4 * y_size)) * sigma;
381  }
382  if (distirbution == "chi") {
383  auto alpha =
384  1. / (8 * y_size) - 1. / (128 * y_size * y_size); // series expansion for gamma relation
385  auto mean = std::sqrt(2) * std::sqrt(y_size) *
386  (1 - alpha); // using Gamma(k+1/2)/Gamma(k) asymptotic formula
387  auto standard_deviation = std::sqrt(2 * y_size * alpha * (2 - alpha));
388  if (sigma == 0) {
389  return mean + n_sigma * standard_deviation;
390  }
391  return (mean + n_sigma * standard_deviation) * sigma;
392  }

445 Vector<t_real> fit_fwhm(const Image<t_real> &psf, const t_int &size) {
446  /*
447  Find FWHM of point spread function, using least squares.
448 
449  psf:: point spread function, assumed to be normalised.
450 
451  The method assumes that you have sampled at least
452  3 pixels across the beam.
453  */
454 
455  auto x0 = std::floor(psf.cols() * 0.5);
456  auto y0 = std::floor(psf.rows() * 0.5);
457 
458  // finding patch
459  Image<t_real> patch =
460  psf.block(std::floor(y0 - size * 0.5) + 1, std::floor(x0 - size * 0.5) + 1, size, size);
461  PURIFY_LOW_LOG("Fitting to Patch");
462 
463  // finding values for least squares
464 
465  std::vector<t_tripletList> entries;
466  auto total_entries = 0;
467 
468  for (t_int i = 0; i < patch.cols(); ++i) {
469  for (t_int j = 0; j < patch.rows(); ++j) {
470  entries.emplace_back(j, i, std::log(patch(j, i)));
471  total_entries++;
472  }
473  }
474  Matrix<t_real> A = Matrix<t_real>::Zero(total_entries, 4);
475  Vector<t_real> q = Vector<t_real>::Zero(total_entries);
476  // putting values into a vector and matrix for least squares
477  for (t_int i = 0; i < total_entries; ++i) {
478  t_real x = entries.at(i).col() - size * 0.5 + 0.5;
479  t_real y = entries.at(i).row() - size * 0.5 + 0.5;
480  A(i, 0) = static_cast<t_real>(x * x); // x^2
481  A(i, 1) = static_cast<t_real>(x * y); // x * y
482  A(i, 2) = static_cast<t_real>(y * y); // y^2
483  q(i) = std::real(entries.at(i).value());
484  }
485  // least squares fitting
486  const auto solution =
487  static_cast<Vector<t_real>>(A.jacobiSvd(Eigen::ComputeThinU | Eigen::ComputeThinV).solve(q));
488  t_real const a = -solution(0);
489  t_real const b = +solution(1) * 0.5;
490  t_real const c = -solution(2);
491  // parameters of Gaussian
492  t_real theta = std::atan2(b, std::sqrt(std::pow(2 * b, 2) + std::pow(c - a, 2)) +
493  (c - a) * 0.5); // some relatively complicated trig identity to
494  // go from tan(2theta) to tan(theta).
495  t_real t = 0.;
496  t_real s = 0.;
497  if (std::abs(b) < 1e-13) {
498  t = a;
499  s = c;
500  theta = 0;
501  } else {
502  t = (a + c - 2 * b / std::sin(2 * theta)) * 0.5;
503  s = (a + c + 2 * b / std::sin(2 * theta)) * 0.5;
504  }
505 
506  t_real const sigma_x = std::sqrt(1 / (2 * t));
507  t_real const sigma_y = std::sqrt(1 / (2 * s));
508 
509  Vector<t_real> fit_parameters = Vector<t_real>::Zero(3);
510 
511  // fit for the beam rms, used for FWHM
512  fit_parameters(0) = sigma_x;
513  fit_parameters(1) = sigma_y;
514  fit_parameters(2) = theta; // because 2*theta is periodic with pi.
515  return fit_parameters;
516 }

544  const Vector<t_complex> &weights, const t_real &oversample_factor,
545  const std::string &weighting_type, const t_real &R,
546  const t_int &ftsizeu, const t_int &ftsizev) {
547  /*
548  Calculate the weights to be applied to the visibilities in the measurement operator.
549  It does none, whiten, natural, uniform, and robust.
550  */
551  if (weighting_type == "none") return weights.array() * 0 + 1.;
552  if (weighting_type == "natural" or weighting_type == "whiten") return weights;
553  Vector<t_complex> out_weights(weights.size());
554  if ((weighting_type == "uniform") or (weighting_type == "robust")) {
555  t_real scale =
556  1. / oversample_factor; // scale for fov, controlling the region of sidelobe supression
557  Matrix<t_complex> gridded_weights = Matrix<t_complex>::Zero(ftsizev, ftsizeu);
558  for (t_int i = 0; i < weights.size(); ++i) {
559  t_int q = utilities::mod(floor(u(i) * scale), ftsizeu);
560  t_int p = utilities::mod(floor(v(i) * scale), ftsizev);
561  gridded_weights(p, q) += 1 * 1; // I get better results assuming all the weights are the
562  // same. It looks like miriad does this as well.
563  }
564  t_complex const sum_weights = (weights.array() * weights.array()).sum();
565  t_complex const sum_grid_weights2 = (gridded_weights.array() * gridded_weights.array()).sum();
566  t_complex const robust_scale =
567  sum_weights / sum_grid_weights2 * 25. *
568  std::pow(10, -2 * R); // Following standard formula, a bit different from miriad.
569  gridded_weights = gridded_weights.array().sqrt();
570  for (t_int i = 0; i < weights.size(); ++i) {
571  t_int q = utilities::mod(floor(u(i) * scale), ftsizeu);
572  t_int p = utilities::mod(floor(v(i) * scale), ftsizev);
573  if (weighting_type == "uniform") out_weights(i) = weights(i) / gridded_weights(p, q);
574  if (weighting_type == "robust") {
575  out_weights(i) = weights(i) / std::sqrt(1. + robust_scale * gridded_weights(p, q) *
576  gridded_weights(p, q));
577  }
578  }
579  } else
580  throw std::runtime_error("Wrong weighting type, " + weighting_type + " not recognised.");
581  return out_weights;
582 }

608 Matrix<t_complex> re_sample_ft_grid(const Matrix<t_complex> &input, const t_real &re_sample_ratio) {
609  /*
610  up samples image using by zero padding the fft
611 
612  input:: fft to be upsampled, with zero frequency at (0,0) of the matrix.
613 
614  */
615  if (re_sample_ratio == 1) return input;
616 
617  // sets up dimensions for old image
618  t_int old_x_centre_floor = std::floor(input.cols() * 0.5);
619  t_int old_y_centre_floor = std::floor(input.rows() * 0.5);
620  // need ceilling in case image is of odd dimension
621  t_int old_x_centre_ceil = std::ceil(input.cols() * 0.5);
622  t_int old_y_centre_ceil = std::ceil(input.rows() * 0.5);
623 
624  // sets up dimensions for new image
625  t_int new_x = std::floor(input.cols() * re_sample_ratio);
626  t_int new_y = std::floor(input.rows() * re_sample_ratio);
627 
628  t_int new_x_centre_floor = std::floor(new_x * 0.5);
629  t_int new_y_centre_floor = std::floor(new_y * 0.5);
630  // need ceilling in case image is of odd dimension
631  t_int new_x_centre_ceil = std::ceil(new_x * 0.5);
632  t_int new_y_centre_ceil = std::ceil(new_y * 0.5);
633 
634  Matrix<t_complex> output = Matrix<t_complex>::Zero(new_y, new_x);
635 
636  t_int box_dim_x;
637  t_int box_dim_y;
638 
639  // now have to move each quadrant into new grid
640  box_dim_x = std::min(old_x_centre_ceil, new_x_centre_ceil);
641  box_dim_y = std::min(old_y_centre_ceil, new_y_centre_ceil);
642  //(0, 0)
643  output.bottomLeftCorner(box_dim_y, box_dim_x) = input.bottomLeftCorner(box_dim_y, box_dim_x);
644 
645  box_dim_x = std::min(old_x_centre_ceil, new_x_centre_ceil);
646  box_dim_y = std::min(old_y_centre_floor, new_y_centre_floor);
647  //(y0, 0)
648  output.topLeftCorner(box_dim_y, box_dim_x) = input.topLeftCorner(box_dim_y, box_dim_x);
649 
650  box_dim_x = std::min(old_x_centre_floor, new_x_centre_floor);
651  box_dim_y = std::min(old_y_centre_ceil, new_y_centre_ceil);
652  //(0, x0)
653  output.bottomRightCorner(box_dim_y, box_dim_x) = input.bottomRightCorner(box_dim_y, box_dim_x);
654 
655  box_dim_x = std::min(old_x_centre_floor, new_x_centre_floor);
656  box_dim_y = std::min(old_y_centre_floor, new_y_centre_floor);
657  //(y0, x0)
658  output.topRightCorner(box_dim_y, box_dim_x) = input.topRightCorner(box_dim_y, box_dim_x);
659 
660  return output;
661 }
662 Matrix<t_complex> re_sample_image(const Matrix<t_complex> &input, const t_real &re_sample_ratio) {
663  const auto ft_plan = operators::fftw_plan::estimate;
664  auto fft =
665  purify::operators::init_FFT_2d<Vector<t_complex>>(input.rows(), input.cols(), 1., ft_plan);
666  Vector<t_complex> ft_grid = Vector<t_complex>::Zero(input.size());
667  std::get<0>(fft)(ft_grid, Vector<t_complex>::Map(input.data(), input.size()));
668  Matrix<t_complex> const new_ft_grid = utilities::re_sample_ft_grid(
669  Matrix<t_complex>::Map(ft_grid.data(), input.rows(), input.cols()), re_sample_ratio);
670  Vector<t_complex> output = Vector<t_complex>::Zero(input.size());
671  fft = purify::operators::init_FFT_2d<Vector<t_complex>>(new_ft_grid.rows(), new_ft_grid.cols(),
672  1., ft_plan);
673  std::get<1>(fft)(output, Vector<t_complex>::Map(new_ft_grid.data(), new_ft_grid.size()));
674  return Matrix<t_complex>::Map(output.data(), new_ft_grid.rows(), new_ft_grid.cols()) *
675  re_sample_ratio;
676 }
