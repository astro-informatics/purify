125 template <class T>
126 Matrix<T> table_column(::casacore::Table const &table, std::string const &column,
127  std::string const &filter, std::integral_constant<bool, true> const &) {
128  if(table.nrow() == 0)
129  return Matrix<T>::Zero(0, 1);
130  auto const taql_table = ::casacore::tableCommand(
131  "USING STYLE PYTHON SELECT " + column + " as R FROM $1 " + filter, table);
132  auto const vtable = taql_table.table();
133  if(vtable.nrow() == 0)
134  return Matrix<T>(0, 1);
135 
136  switch(vtable.tableDesc().columnDesc("R").trueDataType()) {
137 #define PURIFY_MACRO(NAME, TYPE) \
138  case ::casacore::Tp##NAME: \
139  return details::get_taql_scalar<::casacore::TYPE>(vtable).template cast<T>();
140  PURIFY_MACRO(Complex, Complex);
141  PURIFY_MACRO(DComplex, DComplex);
142 #undef PURIFY_MACRO
143 #define PURIFY_MACRO(NAME, TYPE) \
144  case ::casacore::TpArray##NAME: \
145  return details::get_taql_array<::casacore::TYPE>(vtable).template cast<T>();
146  PURIFY_MACRO(Complex, Complex);
147  PURIFY_MACRO(DComplex, DComplex);
148 #undef PURIFY_MACRO
149  default:
150  break;
151  }
152 
153  throw std::runtime_error("Array type is not handled");
154  return Matrix<T>::Zero(0, 1);
155 }
