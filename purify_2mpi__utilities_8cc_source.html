15 void regroup(vis_params &uv_params, std::vector<t_int> &image_index,
16  std::vector<t_int> const &groups_, const t_int max_groups) {
17  std::vector<t_int> groups = groups_;
18  // Figure out size of each group
19  std::map<t_int, t_int> sizes;
20  for (auto g = 0; g < max_groups; g++) sizes[g] = 0;
21  for (auto const &group : groups) ++sizes[group];
22 
23  std::map<t_int, t_int> indices, ends;
24  auto i = 0;
25  for (auto const &item : sizes) {
26  indices[item.first] = i;
27  i += item.second;
28  ends[item.first] = i;
29  }
30  const auto minmax = std::minmax_element(ends.begin(), ends.end());
31  if (std::get<1>(*std::get<0>(minmax)) < 0)
32  throw std::runtime_error("segment end " + std::to_string(std::get<1>(*std::get<0>(minmax))) +
33  " less than 0. Not a valid end.");
34  if (std::get<1>(*std::get<1>(minmax)) > uv_params.size())
35  throw std::runtime_error("segment end " + std::to_string(std::get<1>(*std::get<1>(minmax))) +
36  " larger than data vector " +
37  std::to_string(static_cast<t_int>(uv_params.size())) +
38  ". End not valid.");
39 
40  auto const expected = [&ends](t_int i) {
41  t_int j = 0;
42  for (auto const end : ends) {
43  if (i < end.second) return j;
44  ++j;
45  }
46  return j;
47  };
48 
49  i = 0;
50  while (i < uv_params.u.size()) {
51  auto const expected_proc = expected(i);
52  if (groups[i] == expected_proc) {
53  ++i;
54  continue;
55  }
56  auto &swapper = indices[groups[i]];
57  if (groups[swapper] == expected(swapper)) {
58  ++swapper;
59  continue;
60  }
61  if (swapper >= uv_params.u.size())
62  throw std::runtime_error("regroup (groups, " + std::to_string(groups[swapper]) + ", " +
63  std::to_string(groups[i]) + ") index out of bounds " +
64  std::to_string(i) + " " + std::to_string(swapper) +
65  " >= " + std::to_string(uv_params.u.size()));
66  std::swap(groups[i], groups[swapper]);
67  std::swap(uv_params.u(i), uv_params.u(swapper));
68  std::swap(uv_params.v(i), uv_params.v(swapper));
69  std::swap(uv_params.w(i), uv_params.w(swapper));
70  std::swap(uv_params.vis(i), uv_params.vis(swapper));
71  std::swap(uv_params.weights(i), uv_params.weights(swapper));
72  std::swap(image_index[i], image_index[swapper]);
73 
74  ++swapper;
75  }
76 }
