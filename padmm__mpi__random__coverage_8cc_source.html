76 std::shared_ptr<sopt::algorithm::ImagingProximalADMM<t_complex>> padmm_factory(
77  std::shared_ptr<sopt::LinearTransform<Vector<t_complex>> const> const &measurements,
78  const sopt::wavelets::SARA &sara, const Image<t_complex> &ground_truth_image,
79  const utilities::vis_params &uv_data, const t_real sigma, const sopt::mpi::Communicator &comm) {
80  auto const Psi = sopt::linear_transform<t_complex>(sara, ground_truth_image.rows(),
81  ground_truth_image.cols(), comm);
82 
83 #if PURIFY_PADMM_ALGORITHM == 2
84  auto const epsilon = std::sqrt(
85  comm.all_sum_all(std::pow(utilities::calculate_l2_radius(uv_data.vis.size(), sigma), 2)));
86 #elif PURIFY_PADMM_ALGORITHM == 3 || PURIFY_PADMM_ALGORITHM == 1
87  auto const epsilon = utilities::calculate_l2_radius(uv_data.vis.size(), sigma);
88 #endif
89  PURIFY_LOW_LOG("SARA Size = {}, Rank = {}", sara.size(), comm.rank());
90  const t_real gamma =
91  utilities::step_size(uv_data.vis, measurements,
92  std::make_shared<sopt::LinearTransform<Vector<t_complex>> const>(Psi),
93  sara.size()) *
94  1e-3;
95  PURIFY_LOW_LOG("Epsilon {}, Rank = {}", epsilon, comm.rank());
96  PURIFY_LOW_LOG("Gamma {}, SARA Size = {}, Rank = {}", gamma, sara.size(), comm.rank());
97 
98  // shared pointer because the convergence function need access to some data that we would rather
99  // not reproduce. E.g. padmm definition is self-referential.
100  auto padmm = std::make_shared<sopt::algorithm::ImagingProximalADMM<t_complex>>(uv_data.vis);
101  padmm->itermax(50)
102  .gamma(comm.all_reduce(gamma, MPI_MAX))
103  .relative_variation(1e-3)
104  .l2ball_proximal_epsilon(epsilon)
105 #if PURIFY_PADMM_ALGORITHM == 2
106  // communicator ensuring l2 norm in l2ball proximal is global
107  .l2ball_proximal_communicator(comm)
108 #endif
109  // communicator ensuring l1 norm in l1 proximal is global
110  .l1_proximal_adjoint_space_comm(comm)
111  .tight_frame(false)
112  .l1_proximal_tolerance(1e-2)
113  .l1_proximal_nu(1)
114  .l1_proximal_itermax(50)
115  .l1_proximal_positivity_constraint(true)
116  .l1_proximal_real_constraint(true)
117  .residual_tolerance(epsilon)
118  .lagrange_update_scale(0.9)
119  .nu(1e0)
120  .Psi(Psi)
121  .Phi(*measurements);
122  sopt::ScalarRelativeVariation<t_complex> conv(padmm->relative_variation(),
123  padmm->relative_variation(), "Objective function");
124  std::weak_ptr<decltype(padmm)::element_type> const padmm_weak(padmm);
125  padmm->residual_convergence([padmm_weak, conv, comm](
126  Vector<t_complex> const &x,
127  Vector<t_complex> const &residual) mutable -> bool {
128  auto const padmm = padmm_weak.lock();
129 #if PURIFY_PADMM_ALGORITHM == 2
130  auto const residual_norm = sopt::mpi::l2_norm(residual, padmm->l2ball_proximal_weights(), comm);
131  auto const result = residual_norm < padmm->residual_tolerance();
132 #elif PURIFY_PADMM_ALGORITHM == 3 || PURIFY_PADMM_ALGORITHM == 1
133  auto const residual_norm = sopt::l2_norm(residual, padmm->l2ball_proximal_weights());
134  auto const result =
135  comm.all_reduce<int8_t>(residual_norm < padmm->residual_tolerance(), MPI_LAND);
136 #endif
137  SOPT_LOW_LOG(" - [PADMM] Residuals: {} <? {}", residual_norm, padmm->residual_tolerance());
138  return result;
139  });
140 
141  padmm->objective_convergence([padmm_weak, conv, comm](Vector<t_complex> const &x,
142  Vector<t_complex> const &) mutable -> bool {
143  auto const padmm = padmm_weak.lock();
144 #if PURIFY_PADMM_ALGORITHM == 2
145  return conv(sopt::mpi::l1_norm(padmm->Psi().adjoint() * x, padmm->l1_proximal_weights(), comm));
146 #elif PURIFY_PADMM_ALGORITHM == 3 || PURIFY_PADMM_ALGORITHM == 1
147  return comm.all_reduce<uint8_t>(
148  conv(sopt::l1_norm(padmm->Psi().adjoint() * x, padmm->l1_proximal_weights())), MPI_LAND);
149 #endif
150  });
151 
152  return padmm;
153 }
