72 t_real calc_for_pswf(const t_real eta0, const t_real J, const t_real alpha) {
73  // polynomial coefficients for prolate spheriodal wave function rational approximation
74  const std::array<t_real, 6> p1 = {
75  {8.203343e-2, -3.644705e-1, 6.278660e-1, -5.335581e-1, 2.312756e-1, 2 * 0.0}};
76  const std::array<t_real, 6> p2 = {
77  {4.028559e-3, -3.697768e-2, 1.021332e-1, -1.201436e-1, 6.412774e-2, 2 * 0.0}};
78  const std::array<t_real, 3> q1 = {{1., 8.212018e-1, 2.078043e-1}};
79  const std::array<t_real, 3> q2 = {{1., 9.599102e-1, 2.918724e-1}};
80 
81  if (J != 6 or alpha != 1) {
82  return 0;
83  }
84  // Calculating numerator and denominator using Horner's rule.
85  // PSWF = numerator / denominator
86 
87  auto fraction = [](t_real eta, std::array<t_real, 6> const &p, std::array<t_real, 3> const &q) {
88  auto const p_size = sizeof(p) / sizeof(p[0]) - 1;
89  auto const q_size = sizeof(q) / sizeof(q[0]) - 1;
90 
91  auto numerator = p[p_size];
92  for (auto i = decltype(p_size){1}; i <= p_size; ++i)
93  numerator = eta * numerator + p[p_size - i];
94 
95  auto denominator = q[q_size];
96  for (auto i = decltype(q_size){1}; i <= q_size; ++i)
97  denominator = eta * denominator + q[q_size - i];
98 
99  return numerator / denominator;
100  };
101  if (0 <= std::abs(eta0) and std::abs(eta0) <= 0.75)
102  return fraction(eta0 * eta0 - 0.75 * 0.75, p1, q1);
103  if (0.75 < std::abs(eta0) and std::abs(eta0) <= 1) return fraction(eta0 * eta0 - 1 * 1, p2, q2);
104 
105  return 0;
106 }

158 t_real kernel_linear_interp(const Vector<t_real> &samples, const t_real x, const t_real J) {
159  /*
160  Calculates kernel using linear interpolation between pre-calculated samples. (see Rapid
161  gridding reconstruction with a minimal oversampling ratio, Beatty et. al. 2005)
162  */
163  t_int total_samples = samples.size();
164 
165  t_real i_effective = (x + J / 2) * total_samples / J;
166 
167  t_real i_0 = floor(i_effective);
168  t_real i_1 = ceil(i_effective);
169  // case where i_effective is a sample point
170  if (std::abs(i_0 - i_1) == 0) {
171  return samples(i_0);
172  }
173  // linearly interpolate from nearest neighbour
174  t_real y_0;
175  t_real y_1;
176  if (i_0 < 0 or i_0 >= total_samples) {
177  y_0 = 0;
178  } else {
179  y_0 = samples(i_0);
180  }
181  if (i_1 < 0 or i_1 >= total_samples) {
182  y_1 = 0;
183  } else {
184  y_1 = samples(i_1);
185  }
186  t_real output = y_0 + (y_1 - y_0) / (i_1 - i_0) * (i_effective - i_0);
187  return output;
188 }
