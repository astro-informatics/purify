148 template <class T>
149 std::tuple<sopt::OperatorFunction<T>, sopt::OperatorFunction<T>> init_zero_padding_2d(
150  const Image<typename T::Scalar> &S, const t_real &oversample_ratio) {
151  const t_uint imsizex_ = S.cols();
152  const t_uint imsizey_ = S.rows();
153  const t_uint ftsizeu_ = std::floor(imsizex_ * oversample_ratio);
154  const t_uint ftsizev_ = std::floor(imsizey_ * oversample_ratio);
155  const t_uint x_start = std::floor(ftsizeu_ * 0.5 - imsizex_ * 0.5);
156  const t_uint y_start = std::floor(ftsizev_ * 0.5 - imsizey_ * 0.5);
157  auto direct = [=](T &output, const T &x) {
158  assert(x.size() == imsizex_ * imsizey_);
159  output = Vector<t_complex>::Zero(ftsizeu_ * ftsizev_);
160 #pragma omp parallel for collapse(2)
161  for (t_uint j = 0; j < imsizey_; j++) {
162  for (t_uint i = 0; i < imsizex_; i++) {
163  const t_uint input_index = utilities::sub2ind(j, i, imsizey_, imsizex_);
164  const t_uint output_index =
165  utilities::sub2ind(y_start + j, x_start + i, ftsizev_, ftsizeu_);
166  output(output_index) = S(j, i) * x(input_index);
167  }
168  }
169  };
170  auto indirect = [=](T &output, const T &x) {
171  assert(x.size() == ftsizeu_ * ftsizev_);
172  output = T::Zero(imsizey_ * imsizex_);
173 #pragma omp parallel for collapse(2)
174  for (t_uint j = 0; j < imsizey_; j++) {
175  for (t_uint i = 0; i < imsizex_; i++) {
176  const t_uint output_index = utilities::sub2ind(j, i, imsizey_, imsizex_);
177  const t_uint input_index = utilities::sub2ind(y_start + j, x_start + i, ftsizev_, ftsizeu_);
178  output(output_index) = S(j, i) * x(input_index);
179  }
180  }
181  };
182  return std::make_tuple(direct, indirect);
183 }

209 template <class T>
210 std::tuple<sopt::OperatorFunction<T>, sopt::OperatorFunction<T>> init_FFT_2d(
211  const t_uint &imsizey_, const t_uint &imsizex_, const t_real &oversample_factor_,
212  const fftw_plan fftw_plan_flag_ = fftw_plan::measure) {
213  t_int const ftsizeu_ = std::floor(imsizex_ * oversample_factor_);
214  t_int const ftsizev_ = std::floor(imsizey_ * oversample_factor_);
215  t_int plan_flag = (FFTW_MEASURE | FFTW_PRESERVE_INPUT);
216  switch (fftw_plan_flag_) {
217  case (fftw_plan::measure):
218  plan_flag = (FFTW_MEASURE | FFTW_PRESERVE_INPUT);
219  break;
220  case (fftw_plan::estimate):
221  plan_flag = (FFTW_ESTIMATE | FFTW_PRESERVE_INPUT);
222  break;
223  }
224 
225 #ifdef PURIFY_OPENMP_FFTW
226  PURIFY_LOW_LOG("Using OpenMP threading with FFTW.");
227  fftw_init_threads();
228  fftw_plan_with_nthreads(omp_get_max_threads());
229 #endif
230  Vector<typename T::Scalar> src = Vector<t_complex>::Zero(ftsizev_ * ftsizeu_);
231  Vector<typename T::Scalar> dst = Vector<t_complex>::Zero(ftsizev_ * ftsizeu_);
232  // creating plans
233  const auto del = [](fftw_plan_s *plan) { fftw_destroy_plan(plan); };
234  const std::shared_ptr<fftw_plan_s> m_plan_forward(
235  fftw_plan_dft_2d(ftsizev_, ftsizeu_, reinterpret_cast<fftw_complex *>(src.data()),
236  reinterpret_cast<fftw_complex *>(dst.data()), FFTW_FORWARD, plan_flag),
237  del);
238  const std::shared_ptr<fftw_plan_s> m_plan_inverse(
239  fftw_plan_dft_2d(ftsizev_, ftsizeu_, reinterpret_cast<fftw_complex *>(src.data()),
240  reinterpret_cast<fftw_complex *>(dst.data()), FFTW_BACKWARD, plan_flag),
241  del);
242  auto const direct = [m_plan_forward, ftsizeu_, ftsizev_](T &output, const T &input) {
243  assert(input.size() == ftsizev_ * ftsizeu_);
244  output = Matrix<typename T::Scalar>::Zero(input.rows(), input.cols());
245  fftw_execute_dft(
246  m_plan_forward.get(),
247  const_cast<fftw_complex *>(reinterpret_cast<const fftw_complex *>(input.data())),
248  reinterpret_cast<fftw_complex *>(output.data()));
249  output /= std::sqrt(output.size());
250  };
251  auto const indirect = [m_plan_inverse, ftsizeu_, ftsizev_](T &output, const T &input) {
252  assert(input.size() == ftsizev_ * ftsizeu_);
253  output = Matrix<typename T::Scalar>::Zero(input.rows(), input.cols());
254  fftw_execute_dft(
255  m_plan_inverse.get(),
256  const_cast<fftw_complex *>(reinterpret_cast<const fftw_complex *>(input.data())),
257  reinterpret_cast<fftw_complex *>(output.data()));
258  output /= std::sqrt(output.size());
259  };
260  return std::make_tuple(direct, indirect);
261 }

373 template <class T>
374 std::shared_ptr<sopt::LinearTransform<T> const> init_degrid_operator_2d(
375  const Vector<t_real> &u, const Vector<t_real> &v, const Vector<t_real> &w,
376  const Vector<t_complex> &weights, const t_uint &imsizey, const t_uint &imsizex,
377  const t_real &oversample_ratio = 2, const t_uint &power_iters = 100,
378  const t_real &power_tol = 1e-4, const kernels::kernel kernel = kernels::kernel::kb,
379  const t_uint Ju = 4, const t_uint Jv = 4, const bool w_term = false, const t_uint Jw = 6,
380  const t_real &cellx = 1, const t_real &celly = 1) {
381  const operators::fftw_plan ft_plan = operators::fftw_plan::measure;
382  std::array<t_int, 3> N = {0, 1, static_cast<t_int>(imsizey * imsizex)};
383  std::array<t_int, 3> M = {0, 1, static_cast<t_int>(u.size())};
384  sopt::OperatorFunction<T> directDegrid, indirectDegrid;
385  std::tie(directDegrid, indirectDegrid) = purify::operators::base_degrid_operator_2d<T>(
386  u, v, w, weights, imsizey, imsizex, oversample_ratio, kernel, Ju, Jv, ft_plan, w_term, Jw,
387  cellx, celly);
388  auto direct = directDegrid;
389  auto indirect = indirectDegrid;
390  const t_real op_norm = sopt::algorithm::power_method<T>({direct, M, indirect, N}, power_iters,
391  power_tol, T::Ones(imsizex * imsizey));
392  auto operator_norm = purify::operators::init_normalise<T>(op_norm);
393  direct = sopt::chained_operators<T>(direct, operator_norm);
394  indirect = sopt::chained_operators<T>(operator_norm, indirect);
395  return std::make_shared<sopt::LinearTransform<T> const>(direct, M, indirect, N);
396 }

515  const std::vector<std::shared_ptr<sopt::LinearTransform<T>>> &measure_op,
516  const std::vector<std::tuple<t_uint, t_uint>> seg) {

562 template <class T, class... ARGS>
563 std::shared_ptr<sopt::LinearTransform<T> const> init_super_operator(
564  const std::function<std::shared_ptr<sopt::LinearTransform<T> const>> &create_operator,
565  const utilities::vis_params &uv_vis_data, const t_uint &seg_num, ARGS &&... args) {
566  if (seg_num == 0) throw std::runtime_error("Number of segments = 0");
567  std::vector<std::tuple<t_uint, t_uint>> seg(seg_num);
568  const t_uint segment =
569  std::floor(static_cast<t_real>(uv_vis_data.size()) / static_cast<t_real>(seg_num));
570  t_uint total = 0;
571  for (t_uint i = 0; i < seg_num; i++) {
572  const t_uint delta = (i == (seg_num - 1)) ? (uv_vis_data.size() - total) : segment;
573  seg.emplace_back(total, delta);
574  total += delta;
575  }
576  if (total != uv_vis_data.size()) throw std::runtime_error("Segments do not sum to the total");
577  return init_super_operator(create_operator, uv_vis_data, seg, std::forward<ARGS>(args)...);
578 }
