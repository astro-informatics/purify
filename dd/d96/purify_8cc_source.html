143 std::shared_ptr<sopt::algorithm::ImagingProximalADMM<t_complex>> padmm_factory(
144  std::shared_ptr<sopt::LinearTransform<Vector<t_complex>> const> const &measurements,
145  std::shared_ptr < sopt::LinearTransform<Vector<t_complex>> const &psi,
146  const utilities::vis_params &uv_data, const sopt::mpi::Communicator &comm) {
147 #if PURIFY_PADMM_ALGORITHM == 2
148  auto const epsilon =
149  std::sqrt(comm.all_sum_all(std::pow(utilities::calculate_l2_radius(uv_data.vis, sigma), 2)));
150 #elif PURIFY_PADMM_ALGORITHM == 3 || PURIFY_PADMM_ALGORITHM == 1
151  auto const epsilon = utilities::calculate_l2_radius(uv_data.vis, sigma);
152 #endif
153  auto const gamma =
154  (psi->adjoint() * (measurements->adjoint() * uv_data.vis)).cwiseAbs().maxCoeff() * 1e-3;
155  PURIFY_MEDIUM_LOG("Epsilon {}", epsilon);
156  PURIFY_MEDIUM_LOG("Gamma {}", gamma);
157 
158  // shared pointer because the convergence function need access to some data that we would rather
159  // not reproduce. E.g. padmm definition is self-referential.
160  auto padmm = std::make_shared<sopt::algorithm::ImagingProximalADMM<t_complex>>(uv_data.vis);
161  padmm->itermax(50)
162  .gamma(comm.all_reduce(gamma, MPI_MAX))
163  .relative_variation(1e-3)
164  .l2ball_proximal_epsilon(epsilon)
165 #if PURIFY_PADMM_ALGORITHM == 2
166  // communicator ensuring l2 norm in l2ball proximal is global
167  .l2ball_proximal_communicator(comm)
168 #endif
169  // communicator ensuring l1 norm in l1 proximal is global
170  .l1_proximal_adjoint_space_comm(comm)
171  .tight_frame(false)
172  .l1_proximal_tolerance(1e-2)
173  .l1_proximal_nu(1)
174  .l1_proximal_itermax(50)
175  .l1_proximal_positivity_constraint(true)
176  .l1_proximal_real_constraint(true)
177  .residual_tolerance(epsilon)
178  .lagrange_update_scale(0.9)
179  .nu(1e0)
180  .Psi(*psi)
181  .Phi(*measurements);
182  sopt::ScalarRelativeVariation<t_complex> conv(padmm->relative_variation(),
183  padmm->relative_variation(), "Objective function");
184  std::weak_ptr<decltype(padmm)::element_type> const padmm_weak(padmm);
185  padmm->residual_convergence([padmm_weak, conv, comm](
186  Vector<t_complex> const &x,
187  Vector<t_complex> const &residual) mutable -> bool {
188  auto const padmm = padmm_weak.lock();
189 #if PURIFY_PADMM_ALGORITHM == 2
190  auto const residual_norm = sopt::mpi::l2_norm(residual, padmm->l2ball_proximal_weights(), comm);
191  auto const result = residual_norm < padmm->residual_tolerance();
192 #elif PURIFY_PADMM_ALGORITHM == 3 || PURIFY_PADMM_ALGORITHM == 1
193  auto const residual_norm = sopt::l2_norm(residual, padmm->l2ball_proximal_weights());
194  auto const result =
195  comm.all_reduce<int8_t>(residual_norm < padmm->residual_tolerance(), MPI_LAND);
196 #endif
197  SOPT_LOW_LOG(" - [PADMM] Residuals: {} <? {}", residual_norm, padmm->residual_tolerance());
198  return result;
199  });
200 
201  padmm->objective_convergence([padmm_weak, conv, comm](Vector<t_complex> const &x,
202  Vector<t_complex> const &) mutable -> bool {
203  auto const padmm = padmm_weak.lock();
204 #if PURIFY_PADMM_ALGORITHM == 2
205  return conv(sopt::mpi::l1_norm(padmm->Psi().adjoint() * x, padmm->l1_proximal_weights(), comm));
206 #elif PURIFY_PADMM_ALGORITHM == 3 || PURIFY_PADMM_ALGORITHM == 1
207  return comm.all_reduce<uint8_t>(
208  conv(sopt::l1_norm(padmm->Psi().adjoint() * x, padmm->l1_proximal_weights())), MPI_LAND);
209 #endif
210  });
211 
212  return padmm;
213 }

243  assert(params.comm.broadcast(diagnostic.x).isApprox(diagnostic.x));
244 
245  // then writes stuff to files

248  if(params.comm.is_root()) {
