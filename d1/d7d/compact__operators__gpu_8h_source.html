53  u, v, w, weights, imsizey_, imsizex_, oversample_ratio, kernelu, kernelv, kernelw, Ju, Jv, Jw,
54  w_term);

56  interpolation_matrix = purify::compress_outer(interpolation_matrix);

58  Sparse<t_complex> gTg = interpolation_matrix.adjoint() * interpolation_matrix;
59  gTg.prune([&](const t_uint &i, const t_uint &j, const t_complex &value) {
60  return std::abs(value) > 1e-12;
61  });
62  PURIFY_LOW_LOG("GTG non zeros: {}", gTg.nonZeros());
63  const af::array GTG = gpu::init_af_G_2d(gTg);
64  const sopt::OperatorFunction<af::array> directGgpu
65  = [GTG](af::array &output, const af::array &input) { output = af::matmul(GTG, input); };
66  const sopt::OperatorFunction<Vector<t_complex>> directGhost
67  = gpu::host_wrapper(directGgpu, gTg.cols(), gTg.rows());
68  return [=](Vector<t_complex> &output, const Vector<t_complex> &input) {
69  Vector<t_complex> dist_swap;
70  if(comm.is_root()) {
71  assert(input.size() > 0);
72  distributor.scatter(input, dist_swap);
73  } else {
74  distributor.scatter(dist_swap);
75  }
76  Vector<t_complex> dist_out;
77  directGhost(dist_out, dist_swap);
78  if(not comm.is_root()) {
