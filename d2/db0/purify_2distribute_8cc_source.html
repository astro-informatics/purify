67 Vector<t_int> distance_distribution(Vector<t_real> const &u, Vector<t_real> const &v) {
68  // sort visibilities by distance from centre
69  Vector<t_int> index = Vector<t_int>::LinSpaced(u.size(), 0, u.size());
70  std::sort(index.data(), index.data() + index.size(), [&u, &v](int a, int b) {
71  return std::sqrt(u(a) * u(a) + v(a) * v(a)) < std::sqrt(u(b) * u(b) + v(b) * v(b));
72  });
73  return index;

76 Vector<t_int> equal_distribution(Vector<t_real> const &u, Vector<t_real> const &v,
77  t_int const &grid_size) {
78  // distribute visibilities by density calculated from histogram
79  t_real const max_u = u.array().maxCoeff();
80  t_real const max_v = v.array().maxCoeff();
81  t_real const min_u = u.array().minCoeff();
82  t_real const min_v = v.array().minCoeff();
83  // scaling for histogram grid
84  Vector<t_real> const scaled_u = (u.array() - min_u) * (grid_size - 1) / (max_u - min_u);
85  Vector<t_real> const scaled_v = (v.array() - min_v) * (grid_size - 1) / (max_v - min_v);
86  Image<t_int> histogram = Image<t_int>::Zero(grid_size, grid_size);
87  Vector<t_int> index = Vector<t_int>::LinSpaced(u.size(), 0, u.size());
88  // creating histogram grid
89  for (t_int i = 0; i < index.size(); i++) {
90  histogram(std::floor(scaled_u(i)), std::floor(scaled_v(i))) += 1;
91  }
92  // sorting indicies
93  std::sort(index.data(), index.data() + index.size(),
94  [&histogram, &scaled_u, &scaled_v](int a, int b) {
95  return (histogram(std::floor(scaled_u(a)), std::floor(scaled_v(a))) <
96  histogram(std::floor(scaled_u(b)), std::floor(scaled_v(b))));
97  });
98  return index;
99 }
100 std::tuple<std::vector<t_int>, std::vector<t_real>> kmeans_algo(
101  const Vector<t_real> &w, const t_int number_of_nodes, const t_int iters,
102  const std::function<t_real(t_real)> &cost) {
103  std::vector<t_int> w_node(w.size(), 0);
104  std::vector<t_real> w_centre(number_of_nodes, 0);
105  std::vector<t_real> w_sum(number_of_nodes, 0);
106  std::vector<t_real> w_count(number_of_nodes, 0);
107  t_real diff = 1;
108  t_real const wmin = w.minCoeff();
109  t_real const wmax = w.maxCoeff();
110  for (int i = 0; i < w_centre.size(); i++)
111  w_centre[i] = (i * (wmax - wmin) / number_of_nodes + wmin);
112  // lopp through even nodes to reduces w-term
113  for (int n = 0; n < iters; n++) {
114  PURIFY_DEBUG("clustering iteration {}", n);
115  for (int i = 0; i < w.size(); i++) {
116  t_real min = 1e10;
117  for (int node = 0; node < number_of_nodes; node++) {
118  const t_real cost_val = cost(w(i) - w_centre.at(node));
119  if (cost_val < min) {
120  min = cost_val;
121  w_node[i] = node;
122  }
123  }
124  w_sum[w_node[i]] += w(i);
125  w_count[w_node[i]]++;
126  }
127  for (int j = 0; j < number_of_nodes; j++) {
128  diff += std::abs(w_sum.at(j) / w_count.at(j) - w_centre.at(j)) / std::abs(w_centre.at(j));
129  w_centre[j] = (w_count.at(j) > 0) ? w_sum.at(j) / w_count.at(j) : 0;
130  PURIFY_DEBUG("Node {} has {} visibilities, using w-stack w = {}.", j, w_count.at(j),
131  w_centre.at(j));
132  w_sum[j] = 0;
133  w_count[j] = 0;
134  }
135  if ((diff / number_of_nodes) < 1e-3) {
136  PURIFY_DEBUG("Converged!");
137  break;
138  } else {
139  PURIFY_DEBUG("relative_diff = {}", diff);
140  diff = 0;
141  }
142  }
143 
144  return std::make_tuple(w_node, w_centre);
145 }
