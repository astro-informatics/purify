50 template <class STORAGE_INDEX_TYPE = t_int>
51 Sparse<t_complex, STORAGE_INDEX_TYPE> init_gridding_matrix_2d(
52  const t_uint number_of_images, const std::vector<t_int> &image_index, const Vector<t_real> &u,
53  const Vector<t_real> &v, const Vector<t_complex> &weights, const t_uint &imsizey_,
54  const t_uint &imsizex_, const t_real &oversample_ratio,
55  const std::function<t_real(t_real)> kernelu, const std::function<t_real(t_real)> kernelv,
56  const t_uint Ju, const t_uint Jv) {
57  if (std::any_of(image_index.begin(), image_index.end(), [&number_of_images](int index) {
58  return index < 0 or index > (number_of_images - 1);
59  }))
60  throw std::runtime_error("Image index is out of bounds");
61  const t_uint ftsizev_ = std::floor(imsizey_ * oversample_ratio);
62  const t_uint ftsizeu_ = std::floor(imsizex_ * oversample_ratio);
63  const t_uint rows = u.size();
64  const STORAGE_INDEX_TYPE cols = ftsizeu_ * ftsizev_ * number_of_images;
65  if (u.size() != v.size())
66  throw std::runtime_error(
67  "Size of u and v vectors are not the same for creating gridding matrix.");
68 
69  Sparse<t_complex, STORAGE_INDEX_TYPE> interpolation_matrix(rows, cols);
70  interpolation_matrix.reserve(Vector<t_int>::Constant(rows, Ju * Jv));
71 
72  const t_complex I(0, 1);
73  const t_int ju_max = std::min(Ju, ftsizeu_);
74  const t_int jv_max = std::min(Jv, ftsizev_);
75  // If I collapse this for loop there is a crash when using MPI... Sparse<>::insert() doesn't work
76  // right
77 #pragma omp parallel for
78  for (t_int m = 0; m < rows; ++m) {
79  for (t_int ju = 1; ju < ju_max + 1; ++ju) {
80  for (t_int jv = 1; jv < jv_max + 1; ++jv) {
81  const t_real k_u = std::floor(u(m) - ju_max * 0.5);
82  const t_real k_v = std::floor(v(m) - jv_max * 0.5);
83  const t_uint q = utilities::mod(k_u + ju, ftsizeu_);
84  const t_uint p = utilities::mod(k_v + jv, ftsizev_);
85  assert(image_index.at(m) < number_of_images);
86  const STORAGE_INDEX_TYPE index =
87  static_cast<STORAGE_INDEX_TYPE>(utilities::sub2ind(p, q, ftsizev_, ftsizeu_)) +
88  static_cast<STORAGE_INDEX_TYPE>(image_index.at(m)) *
89  static_cast<STORAGE_INDEX_TYPE>(ftsizev_ * ftsizeu_);
90  interpolation_matrix.insert(static_cast<STORAGE_INDEX_TYPE>(m), index) =
91  std::exp(-2 * constant::pi * I * ((k_u + ju) * 0.5 + (k_v + jv) * 0.5)) *
92  kernelu(u(m) - (k_u + ju)) * kernelv(v(m) - (k_v + jv)) * weights(m);
93  }
94  }
95  }
96  return interpolation_matrix;
97 }

100 Sparse<t_complex, STORAGE_INDEX_TYPE> init_gridding_matrix_2d(
101  const t_uint number_of_images, const std::vector<t_int> &image_index,
102  const std::vector<t_real> &w_stacks, const Vector<t_real> &u, const Vector<t_real> &v,
103  const Vector<t_real> &w, const Vector<t_complex> &weights, const t_uint imsizey_,
104  const t_uint imsizex_, const t_real oversample_ratio,
105  const std::function<t_real(t_real)> &ftkerneluv, const std::function<t_real(t_real)> &kerneluv,
106  const t_uint Ju, const t_uint Jw, const t_real cellx, const t_real celly,
107  const t_real abs_error, const t_real rel_error, const dde_type dde) {
108  const t_uint ftsizev_ = std::floor(imsizey_ * oversample_ratio);
109  const t_uint ftsizeu_ = std::floor(imsizex_ * oversample_ratio);
110  const t_real du = widefield::pixel_to_lambda(cellx, imsizex_, oversample_ratio);
111  const t_real dv = widefield::pixel_to_lambda(celly, imsizey_, oversample_ratio);
112  PURIFY_HIGH_LOG("du x du: {}, {}", du, dv);
113  if (std::abs(du - dv) > 1e-6)
114  throw std::runtime_error(
115  "Field of view along l and m is not the same, this assumption is required for "
116  "w-projection.");
117  const t_uint rows = u.size();
118  const STORAGE_INDEX_TYPE cols = ftsizeu_ * ftsizev_ * number_of_images;
119  if (u.size() != v.size())
120  throw std::runtime_error(
121  "Size of u and v vectors are not the same for creating gridding matrix.");
122  if (u.size() != w.size())
123  throw std::runtime_error(
124  "Size of u and w vectors are not the same for creating gridding matrix.");
125  if (u.size() != weights.size())
126  throw std::runtime_error(
127  "Size of u and w vectors are not the same for creating gridding matrix.");
128  if (Ju > Jw)
129  throw std::runtime_error(
130  "w kernel size must be at least the size of w=0 kernel, must have Ju <= Jw.");
131  // count gridding coefficients with variable support size
132  Vector<t_int> total_coeffs = Vector<t_int>::Zero(w.size());
133  for (int i = 0; i < w.size(); i++) {
134  const t_int Ju_max = widefield::w_support(w(i) - w_stacks.at(image_index.at(i)), du,
135  static_cast<t_int>(Ju), static_cast<t_int>(Jw));
136  total_coeffs(i) = Ju_max * Ju_max;
137  }
138  const t_real num_of_coeffs = total_coeffs.array().cast<t_real>().sum();
139  PURIFY_HIGH_LOG("Using {} rows (coefficients per a row {}), and memory of {} MegaBytes",
140  total_coeffs.size(), total_coeffs.array().cast<t_real>().mean(),
141  16. * num_of_coeffs / std::pow(10., 6));
142  Sparse<t_complex, STORAGE_INDEX_TYPE> interpolation_matrix(static_cast<STORAGE_INDEX_TYPE>(rows),
143  cols);
144  try {
145  interpolation_matrix.reserve(total_coeffs);
146  } catch (std::bad_alloc e) {
147  throw std::runtime_error(
148  "Not enough memory for coefficients, choose upper limit on support size Jw.");
149  }
150 
151  const t_complex I(0., 1.);
152 
153  const t_uint max_evaluations = 1e8;
154  const t_real relative_error = rel_error;
155  const t_real absolute_error = abs_error;
156 
157  auto const ftkernel_radial = [&](const t_real l) -> t_real { return ftkerneluv(l); };
158 
159  std::int64_t coeffs_done = 0;
160  std::int64_t total = 0;
161 
162 #pragma omp parallel for
163  for (t_int m = 0; m < rows; ++m) {
164  // w_projection convolution setup
165  const t_real w_val = w(m) - w_stacks.at(image_index.at(m));
166  const t_int Ju_max = widefield::w_support(w_val, du, Ju, Jw);
167  t_uint evaluations = 0;
168  const t_int kwu = std::floor(u(m) - Ju_max * 0.5);
169  const t_int kwv = std::floor(v(m) - Ju_max * 0.5);
170 
171  for (t_int ju = 1; ju < Ju_max + 1; ++ju) {
172  for (t_int jv = 1; jv < Ju_max + 1; ++jv) {
173  const t_uint q = utilities::mod(kwu + ju, ftsizeu_);
174  const t_uint p = utilities::mod(kwv + jv, ftsizev_);
175  const STORAGE_INDEX_TYPE index =
176  static_cast<STORAGE_INDEX_TYPE>(utilities::sub2ind(p, q, ftsizev_, ftsizeu_)) +
177  static_cast<STORAGE_INDEX_TYPE>(image_index.at(m)) *
178  static_cast<STORAGE_INDEX_TYPE>(ftsizev_ * ftsizeu_);
179  interpolation_matrix.insert(static_cast<STORAGE_INDEX_TYPE>(m), index) =
180  std::exp(-2 * constant::pi * I * ((kwu + ju) * 0.5 + (kwv + jv) * 0.5)) * weights(m) *
181  ((dde == dde_type::wkernel_radial)
182  ? ((std::abs(2 * constant::pi * w_val * (std::sqrt(1 - std::pow(0.5 / du, 2))) -
183  1) > 0.01)
184  ? projection_kernels::exact_w_projection_integration_1d(
185  (u(m) - (kwu + ju)), (v(m) - (kwv + jv)), w_val, du, oversample_ratio,
186  ftkernel_radial, max_evaluations, absolute_error, relative_error,
187  (du > 1.) ? integration::method::p : integration::method::h,
188  evaluations)
189  : kerneluv(std::sqrt(std::pow(u(m) - (kwu + ju), 2) +
190  std::pow(v(m) - (kwv + jv), 2))))
191  : projection_kernels::exact_w_projection_integration(
192  (u(m) - (kwu + ju)), (v(m) - (kwv + jv)), w_val, du, dv, oversample_ratio,
193  ftkernel_radial, ftkernel_radial, max_evaluations, absolute_error,
194  relative_error, integration::method::h, evaluations));
195 #pragma omp critical(add_eval)
196  {
197  coeffs_done++;
198  if (num_of_coeffs > 100) {
199  if ((coeffs_done % (static_cast<t_int>(num_of_coeffs / 100)) == 0)) {
200  PURIFY_LOW_LOG(
201  "w = {}, support = {}x{}, coeffs: {} of {}, {}%", w_val, Ju_max, Ju_max,
202  coeffs_done, num_of_coeffs,
203  static_cast<t_real>(coeffs_done) / static_cast<t_real>(num_of_coeffs) * 100.);
204  }
205  }
206  }
207  }
208  }
209  }
210  interpolation_matrix.makeCompressed();
211  return interpolation_matrix;
212 }

330 template <class T>
331 std::tuple<sopt::OperatorFunction<T>, sopt::OperatorFunction<T>> init_zero_padding_2d(
332  const Image<typename T::Scalar> &S, const t_real &oversample_ratio) {
333  const t_uint imsizex_ = S.cols();
334  const t_uint imsizey_ = S.rows();
335  const t_uint ftsizeu_ = std::floor(imsizex_ * oversample_ratio);
336  const t_uint ftsizev_ = std::floor(imsizey_ * oversample_ratio);
337  const t_uint x_start = std::floor(ftsizeu_ * 0.5 - imsizex_ * 0.5);
338  const t_uint y_start = std::floor(ftsizev_ * 0.5 - imsizey_ * 0.5);
339  auto direct = [=](T &output, const T &x) {
340  assert(x.size() == imsizex_ * imsizey_);
341  output = Vector<t_complex>::Zero(ftsizeu_ * ftsizev_);
342 #pragma omp parallel for collapse(2)
343  for (t_uint j = 0; j < imsizey_; j++) {
344  for (t_uint i = 0; i < imsizex_; i++) {
345  const t_uint input_index = utilities::sub2ind(j, i, imsizey_, imsizex_);
346  const t_uint output_index =
347  utilities::sub2ind(y_start + j, x_start + i, ftsizev_, ftsizeu_);
348  output(output_index) = S(j, i) * x(input_index);
349  }
350  }
351  };
352  auto indirect = [=](T &output, const T &x) {
353  assert(x.size() == ftsizeu_ * ftsizev_);
354  output = T::Zero(imsizey_ * imsizex_);
355 #pragma omp parallel for collapse(2)
356  for (t_uint j = 0; j < imsizey_; j++) {
357  for (t_uint i = 0; i < imsizex_; i++) {
358  const t_uint output_index = utilities::sub2ind(j, i, imsizey_, imsizex_);
359  const t_uint input_index = utilities::sub2ind(y_start + j, x_start + i, ftsizev_, ftsizeu_);
360  output(output_index) = std::conj(S(j, i)) * x(input_index);
361  }
362  }
363  };
364  return std::make_tuple(direct, indirect);
365 }

448  };
